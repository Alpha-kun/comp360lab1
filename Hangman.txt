Hangman[level, size]=
Eq[level,0,Polygon[size,4],
	Block[
		Hangman[level-1,size/2.0],
		Move[size/2.0],
		Hangman[level-1,size/2.0],
		Move[size/2.0],
		Turn[Pi/2.0],
		Move[size/2.0],
		Hangman[level-1,size/2.0],
		Turn[Pi],
		Move[size/2.0],
		Turn[3*Pi/2.0],
		Move[size],
		Turn[Pi]
	]
]
